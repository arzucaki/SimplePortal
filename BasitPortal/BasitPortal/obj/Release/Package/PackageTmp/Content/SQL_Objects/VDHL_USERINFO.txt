ALTER VIEW [dbo].[VDHL_USERINFO] AS
SELECT 
	[USER_ID],
	USER_CODE,
	USER_DESCRIPTION,
	USER_PASSWORD,
	USER_ISACTIVE,
	USER_EMAIL,
	CREATEDDATETIME,
	CREATEDUSERID,
	MODIFIEDDATETIME,
	MODIFIEDUSERID,
	ISNULL(AUTHLEVEL,0) AUTHLEVEL,
	USERPRINTER
FROM
	LPLA_USER WITH (NOLOCK)
	LEFT JOIN LDHL_ITSUSER WITH (NOLOCK) ON [USER_ID] = USERID







GO
