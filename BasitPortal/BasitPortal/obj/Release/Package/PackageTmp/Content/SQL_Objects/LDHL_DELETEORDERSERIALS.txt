ALTER proc [dbo].[LDHL_DELETEORDERSERIALS] 
@Order_Code varchar(40),
@User_ID INT = 0
as 
BEGIN
	DECLARE @Ord_ID BIGINT
	
SELECT 
	@Ord_ID = WAOR_ID
FROM
	LWMS_WAREHOUSEORDER (NOLOCK)
WHERE
	WAOR_CODE = @Order_Code
	
DELETE FROM LWMS_SERIALLISTPACK WHERE SLPA_WAREHOUSEORDERID = @Order_Code 
--AND snl_SN IN(SELECT snhi_SerialNo FROM LV_SNHierarchy WHERE snhi_ParentSerialNo = '00186802400407820439')


DELETE FROM LWMS_SERIALLISTLOG WHERE SELL_WAREHOUSEORDERID = @Ord_ID 


END

GO
