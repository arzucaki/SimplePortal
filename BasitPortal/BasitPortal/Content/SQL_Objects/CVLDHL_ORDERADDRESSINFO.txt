ALTER VIEW [dbo].[CVLDHL_ORDERADDRESSINFO] AS
select 
DISTINCT INID_STRINGFIELD10 ord_Code, dbo.translateturkishchars(INID_STRINGFIELD23) CUSTOMER, 
dbo.translateturkishchars(INID_STRINGFIELD28) ADRESS
 from lcom_INTEGRATIONINDETAIL WITH (NOLOCK)
 LEFT JOIN LCOM_INTEGRATIONINHEAD WITH (NOLOCK) ON INIH_ID= [INID_INTEGRATIONHEADERID]
WHERE [INIH_INTEGRATIONTYPEID] = 1
AND INID_STRINGFIELD23!=''
AND [INIH_FINISHDATETIME]> GETDATE()-14



GO
